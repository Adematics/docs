ClusterMetadata 

type: "io.kestra.plugin.gcp.gke.ClusterMetadata"

# Examples

Fetch a gke cluster metadata

id: "cluster_metadata"
type: "io.kestra.plugin.gcp.gke.ClusterMetadata"
name: "gke-metas"
projectId: my-project-id
zone: eu-west-1c
clusterId: my-cluster-id
