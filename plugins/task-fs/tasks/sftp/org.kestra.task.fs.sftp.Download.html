# Download

type: "org.kestra.task.fs.sftp.Download"

Download file from sftp server

# Examples

id: "download"
type: "org.kestra.task.fs.sftp.Download"
host: localhost
port: 6622
username: foo
password: pass
from: "/in/file.txt"

# Inputs

# from

  • Type: string
  • Dynamic: ✔️
  • Required: ❌

The fully-qualified URIs that point to destination path

# host

  • Type: string
  • Dynamic: ✔️
  • Required: ❌

Hostname of the remote server

# keyfile

  • Type: string
  • Dynamic: ✔️
  • Required: ❌
